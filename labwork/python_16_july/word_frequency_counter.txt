"""
Problem 9: Word Frequency Counter 
Topics: Function, Dictionary, String, List 
Accept a sentence from the user. 
Create a function: 
word_frequency(sentence) 
The function should: 
• Remove punctuation.  
• Ignore case sensitivity.  
• Count frequency of every word.  
• Display words in alphabetical order.  
• Print the most repeated word.  
Example: 
Input: 
Python is easy. Python is powerful. 
Output 
easy :1 
is :2 
powerful :1 
python :2 
 
Most Repeated: 
python 
"""
"""
 Word Frequency Counter
Topics: Function, Dictionary, String, List
"""
import string


def word_frequency(sentence):
    # Remove punctuation
    cleaned = sentence.translate(str.maketrans("", "", string.punctuation))
    # Ignore case
    cleaned = cleaned.lower()

    words = cleaned.split()

    freq = {}
    for word in words:
        freq[word] = freq.get(word, 0) + 1

    # Display words in alphabetical order
    for word in sorted(freq):
        print(f"{word} :{freq[word]}")

    # Print the most repeated word
    most_repeated = max(freq, key=freq.get)
    print("\nMost Repeated:")
    print(most_repeated)


# ---- Take and validate user input ----
while True:
    sentence = input("Enter a sentence: ")
    if sentence.strip() != "":
        break
    print("Invalid input! Sentence cannot be empty.")

word_frequency(sentence)
